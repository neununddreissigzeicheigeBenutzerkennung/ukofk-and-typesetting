\newdimen\bulletwidth
\newdimen\bulletswidth

\def\list#1#2{%
  \def\append##1##2{%
    \expandafter\def\expandafter##1\expandafter{##1##2}%
  }%

  \def\curcon{}%
  \def\precon{}%

  \def\conlen##1{%
    \expandafter\def\expandafter\precon\expandafter{\curcon}%
    \append\curcon{##1}%
    \setbox0=\hbox{\curcon}%
    \bulletwidth=\wd0
    \setbox0=\hbox{\precon}%
    \advance\bulletwidth by -\wd0
  }%

  {%

  \def\item##1{%
    \conlen{#1}%
    \ifdim\bulletwidth>\bulletswidth%
      \bulletswidth\bulletwidth%
    \fi%
  }%

  #2%
  {%

  \def\item##1{%
    \noindent#1\vtop{\advance\hsize-\bulletswidth\noindent##1}\par
  }

  #2}}
}
