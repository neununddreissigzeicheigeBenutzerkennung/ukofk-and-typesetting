\newcount\nextparam
\newcount\digitval

\def\alphanumeral#1{{%
  \digitval=#1%
  \ifnum#1>25
    \nextparam=#1%
    \divide\nextparam by 26
    \alphanumeral{\nextparam}%
    \multiply\nextparam by 26
    \advance\digitval by -\nextparam
  \fi
  \advance\digitval by 97
  \char\digitval
}}

\def\ALPHANUMERAL#1{%
  \uppercase{\expandafter\alphanumeral\expandafter{#1}}%
}
