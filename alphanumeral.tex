%
% version 1.0
% This package aims to make numbers easily displayable using letters[a-z] as digits.
% \alphanumeral displays the number which is its first argument using lower case letters as discribed above.
%
\newcount\nextparam
\newcount\digitval
\def\alphanumeral#1{{
	\digitval#1
	\ifnum#1>25
		\nextparam#1
		\divide\nextparam26
		\alphanumeral{
			\nextparam
		}%
		\multiply\nextparam26
		\advance\digitval-\nextparam
	\fi
	\advance\digitval97
	\char\digitval
}}%
